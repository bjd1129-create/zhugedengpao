## 🔴 第1轮：红方初始方案

### 红方方案概述
- **技术栈**：Python 3 + tkinter GUI + A*寻路算法
- **游戏机制**：30x20网格，蛇自动找食物
- **AI策略**：A*寻路到食物，若无路径则四方向试探
- **代码文件**：snake_round1.py

### ✅ 红方自评优点
1. A*寻路能有效找到食物路径
2. 防止180度掉头的方向限制
3. 分数越高速度越快的动态难度
4. 多线程AI决策（不阻塞主循环）

### ❌ 蓝方初审漏洞（严重度：高）
1. **线程安全问题**：`self.snake`在主线程和AI线程同时读写，无锁保护，可能`RuntimeError: deque mutated during iteration`
2. **A*不考虑蛇身**：寻路时将蛇身视为障碍，但没有考虑蛇尾即将消失的位置（蛇移动时尾会缩进一格）
3. **食物生成可能死循环**：`_spawn_food`在蛇占满整个网格时会无限循环
4. **无边界保护**：tkinter的`winfo_children`遍历会误伤其他widget
5. **AI线程无限循环**：`game_over`检查存在竞态，可能在游戏结束后AI线程仍在运行

### 蓝方评分：❌ 不合格（存在致命并发bug）

```python
def _is_path_safe(self, path, look_ahead=10):
    """模拟沿路径移动look_ahead步，检查是否有空间"""
    # 创建虚拟蛇
    virtual_snake = deque(self.snake)
    
    for i, pos in enumerate(path[1:look_ahead+1], 1):
        virtual_snake.appendleft(pos)
        virtual_snake.pop()  # 假设每步都吃不到食物
        
        # 检查是否撞墙或撞自己
        if not (0 <= pos[0] < GRID_WIDTH and 0 <= pos[1] < GRID_HEIGHT):
            return False
        if pos in list(virtual_snake)[1:]:
            return False
        
        # 检查空间是否足够（蛇长 vs 可用格数）
        available = GRID_WIDTH * GRID_HEIGHT - len(virtual_snake)
        if available < 5:  # 剩余空间少于5格判定为危险
            return False
    
    return True

def _safe_direction(self):
    """找最安全的移动方向"""
    head = self.snake[0]
    candidates = []
    
    for d in [(1,0), (-1,0), (0,1), (0,-1)]:
        if d == (-self.direction[0], -self.direction[1]):
            continue
        new_head = (head[0]+d[0], head[1]+d[1])
        
        if not self._is_safe(new_head):
            continue
        
        # 评估该方向的"生存分数"
        score = self._evaluate_survival(new_head)
        candidates.append((score, d))
    
    if candidates:
        return max(candidates, key=lambda x: x[0])[1]
    return self.direction  # 无法移动，保持方向

def _evaluate_survival(self, pos):
    """评估某位置的生存分数（BFS可达空格数）"""
    return self._bfs_accessible(pos)
```

### 最终代码架构
```python
#!/usr/bin/env python3
"""
🐍 贪吃蛇AI - 红蓝对抗最终版 (v10.0)
改进点：
1. 线程安全：AI在主线程运行
2. A*安全增强：考虑蛇尾消失+look_ahead模拟
3. 生存评估：BFS可达空格数
4. 自适应参数：根据蛇长调整look_ahead
5. 性能统计：游戏次数/平均分/最高分
6. 紧急保护：食物满时吃自己
"""

import tkinter as tk
from tkinter import font as tkfont
import random
from collections import deque

CELL_SIZE = 20
GRID_WIDTH = 30
GRID_HEIGHT = 20
CANVAS_W = CELL_SIZE * GRID_WIDTH
CANVAS_H = CELL_SIZE * GRID_HEIGHT

class SnakeGame:
    def __init__(self, master):
        self.master = master
        self.master.title("🐍 贪吃蛇AI - 最终版 v10.0")
        
        self.snake = deque([(GRID_WIDTH//2, GRID_HEIGHT//2)])
        self.direction = (1, 0)
        self.next_direction = (1, 0)
        self.food = self._spawn_food()
        self.score = 0
        self.game_over = False
        self.paused = False
        
        # 性能统计
        self.stats = {'games': 0, 'total': 0, 'best': 0, 'recent': deque(maxlen=50)}
        
        # 画布
        self.canvas = tk.Canvas(master, width=CANVAS_W, height=CANVAS_H, bg='#1a1a2e')
        self.canvas.pack()
        
        # 状态栏
        self.stats_label = tk.Label(master, text=self._stats_text(),
                                   fg='#4ecca3', bg='#16213e', font=('Courier', 10))
        self.stats_label.pack(fill=tk.X)
        
        # 绑定
        master.bind('<Left>', lambda e: self._set_dir((-1, 0)))
        master.bind('<Right>', lambda e: self._set_dir((1, 0)))
        master.bind('<Up>', lambda e: self._set_dir((0, -1)))
        master.bind('<Down>', lambda e: self._set_dir((0, 1)))
        master.bind('<space>', lambda e: self._toggle_pause())
        
        self._draw()
        self._loop()
    
    def _stats_text(self):
        avg = self.stats['total'] / max(1, self.stats['games'])
        return f"局数:{self.stats['games']} | 平均:{avg:.1f} | 最高:{self.stats['best']} | 当前:{self.score}"
    
    def _spawn_food(self):
        for _ in range(GRID_WIDTH * GRID_HEIGHT):
            p = (random.randrange(GRID_WIDTH), random.randrange(GRID_HEIGHT))
            if p not in self.snake:
                return p
        return self.snake[0]  # 满格应急
    
    def _set_dir(self, d):
        if d != (-self.direction[0], -self.direction[1]):
            self.next_direction = d
    
    def _toggle_pause(self):
        self.paused = not self.paused
    
    def _bfs(self, start, forbidden):
        """BFS计算从start可达的空格数"""
        visited = {start}
        queue = [start]
        count = 0
        while queue:
            x, y = queue.pop(0)
            count += 1
            for dx, dy in [(1,0),(-1,0),(0,1),(0,-1)]:
                nx, ny = x+dx, y+dy
                if 0 <= nx < GRID_WIDTH and 0 <= ny < GRID_HEIGHT:
                    if (nx, ny) not in visited and (nx, ny) not in forbidden:
                        visited.add((nx, ny))
                        queue.append((nx, ny))
        return count
    
    def _astar_safe(self, start, goal):
        """A*寻路，结果附带安全性评估"""
        h = lambda a, b: abs(a[0]-b[0]) + abs(a[1]-b[1])
        
        # 构建障碍集（蛇身-尾，因为尾下一秒会消失）
        body = set(self.snake)
        if len(body) > 1:
            body.discard(self.snake[-1])
        
        open_set = {start}
        came_from = {}
        g = {start: 0}
        f = {start: h(start, goal)}
        
        while open_set:
            cur = min(open_set, key=lambda x: f.get(x, float('inf')))
            if cur == goal:
                path = []
                while cur in came_from:
                    path.append(cur)
                    cur = came_from[cur]
                return path[::-1][1:]
            
            open_set.remove(cur)
            for dx, dy in [(1,0),(-1,0),(0,1),(0,-1)]:
                nxt = (cur[0]+dx, cur[1]+dy)
                if not (0 <= nxt[0] < GRID_WIDTH and 0 <= nxt[1] < GRID_HEIGHT):
                    continue
                if nxt in body:
                    continue
                
                tg = g[cur] + 1
                if nxt not in g or tg < g[nxt]:
                    came_from[nxt] = cur
                    g[nxt] = tg
                    f[nxt] = tg + h(nxt, goal)
                    open_set.add(nxt)
        return []
    
    def _simulate_safe(self, path, steps):
        """模拟沿path走steps步是否安全"""
        vs = deque(self.snake)
        for i, pos in enumerate(path[1:steps+1], 1):
            vs.appendleft(pos)
            vs.pop()
            if not (0 <= pos[0] < GRID_WIDTH and 0 <= pos[1] < GRID_HEIGHT):
                return False
            if pos in list(vs)[1:]:
                return False
        return True
    
    def _ai_dir(self):
        head = self.snake[0]
        path = self._astar_safe(head, self.food)
        
        la = 5 if len(self.snake) < GRID_WIDTH*GRID_HEIGHT*0.3 else \
             10 if len(self.snake) < GRID_WIDTH*GRID_HEIGHT*0.6 else 20
        
        if path and self._simulate_safe(path, la):
            return (path[1][0]-head[0], path[1][1]-head[1])
        
        # 不安全，找生存分数最高的方向
        best_score = -1
        best_dir = self.direction
        for d in [(1,0),(-1,0),(0,1),(0,-1)]:
            if d == (-self.direction[0], -self.direction[1]):
                continue
            nh = (head[0]+d[0], head[1]+d[1])
            if not (0 <= nh[0] < GRID_WIDTH and 0 <= nh[1] < GRID_HEIGHT):
                continue
            if nh in self.snake:
                continue
            score = self._bfs(nh, set(self.snake))
            if score > best_score:
                best_score = score
                best_dir = d
        return best_dir
    
    def _move(self):
        self.direction = self.next_direction
        head = self.snake[0]
        nh = (head[0]+self.direction[0], head[1]+self.direction[1])
        
        self.snake.appendleft(nh)
        if nh == self.food:
            self.score += 10
            self.food = self._spawn_food()
        else:
            self.snake.pop()
        
        if not (0 <= nh[0] < GRID_WIDTH and 0 <= nh[1] < GRID_HEIGHT):
            self._over()
            return
        if nh in list(self.snake)[1:]:
            self._over()
            return
    
    def _over(self):
        self.game_over = True
        self.stats['games'] += 1
        self.stats['total'] += self.score
        self.stats['best'] = max(self.stats['best'], self.score)
        self.stats['recent'].append(self.score)
        self.stats_label.config(text=self._stats_text())
        
        self.canvas.create_text(CANVAS_W//2, CANVAS_H//2,
            text=f"游戏结束!\n得分: {self.score}\n按R键重新开始",
```
